178  {
179  USIDR_ = data;
180  // Clear counter overflow before transmission
181  USISR_ = bits::BV8(USIOIF);
182  synchronized
183  {
184  // Note: the following loop means 7 clock cycles per bit transmitted
185  // Hence at 8MHz, that gives a SPI rate > 1MHz, which might be incompatible
186  // with some devices
187  // FIXME add a NOP to ensure 8 clock cycles per bit, hence 1MHz SPI clock
188  while ((USISR_ & bits::BV8(USIOIF)) == 0) USICR_ |= bits::BV8(USITC);
189  }
190  return USIDR_;
191  }
