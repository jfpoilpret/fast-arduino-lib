34 // Device driver guidelines:
35 // - Template on MANAGER, subclass of I2CDevice
36 // - define types aliases PARENT, PROXY and FUTURE
37 // - constructor shall pass one of i2c::Mode i2c::I2C_STANDARD or i2c::I2C_FAST
38 // to inherited constructor, in order tos pecify the BEST mode supported by this driver
39 // - Define FUTURE subclass (inside device class) for every future requiring input (constant, or user-provided)
40 // naming convention: MethodNameFuture
41 // - FUTURE subclass shall have explicit constructor with mandatory input arguments (no default)
42 // - each async API method returns int (error code) and takes a PROXY to specific future as unique argument
43 // - each async API shall have a matching sync API with same name but different prototype:
44 // direct input arguments, reference to output arguments, bool return (true if OK)
45 // each sync API calls the matching async API after creating the proper Future on the stack
46 // then it calls await() or get() on the future before returning.

293  template<uint8_t SIZE_>
294  class SetRamFuture : public FUTURE<void, containers::array<uint8_t, SIZE_ + 1>>
295  {
296  using PARENT = FUTURE<void, containers::array<uint8_t, SIZE_ + 1>>;
297  using IN = typename PARENT::IN;
298 
299  static IN build_array(uint8_t address, const uint8_t (&data)[SIZE_])
300  {
301  IN input;
302  input[0] = static_cast<uint8_t>(address + RAM_START);
303  input.set(1, data);
304  return input;
305  }
306 
307  public:
309  explicit SetRamFuture(uint8_t address, const uint8_t (&data)[SIZE_])
310  : PARENT{build_array(address, data)}
311  {

313  }
314  SetRamFuture(SetRamFuture<SIZE_>&&) = default;
315  SetRamFuture& operator=(SetRamFuture<SIZE_>&&) = default;

317  bool is_input_valid() const
318  {
319  return ((this->get_input()[0] + SIZE_) <= RAM_END);
