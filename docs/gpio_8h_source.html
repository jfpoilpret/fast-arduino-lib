138  {
139  if (mode == PinMode::OUTPUT)
140  TRAIT::DDR |= bits::BV8(BIT);
141  else
142  TRAIT::DDR &= bits::CBV8(BIT);
143  if (value || (mode == PinMode::INPUT_PULLUP))

147  }
