577  template<uint8_t TIMER_NUM_, typename HANDLER_, void (HANDLER_::*CALLBACK_)(uint32_t)> static void rtt_method()
578  {
579  static constexpr board::Timer NTIMER = isr_handler::check_timer<TIMER_NUM_>();
580  auto handler = interrupt::HandlerHolder<RTT<NTIMER>>::handler();
581  handler->on_timer();
582  interrupt::CallbackHandler<void (HANDLER_::*)(uint32_t), CALLBACK_>::call(handler->millis());
583  }
584 
585  template<uint8_t TIMER_NUM_, void (*CALLBACK_)(uint32_t)> static void rtt_function()
586  {
587  static constexpr board::Timer NTIMER = isr_handler::check_timer<TIMER_NUM_>();
588  auto handler = interrupt::HandlerHolder<RTT<NTIMER>>::handler();
589  handler->on_timer();
590  CALLBACK_(handler->millis());
591  }

593  template<uint8_t TIMER_NUM_, typename EVENT_, uint32_t PERIOD_> static void rtt_event()

596  auto handler = interrupt::HandlerHolder<RTT<NTIMER>>::handler();
597  handler->on_timer();
