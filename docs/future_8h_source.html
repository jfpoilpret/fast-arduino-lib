628  bool set_future_value_(uint8_t chunk)
629  {
630  // Check this future is waiting for data
631  if (status_ != FutureStatus::NOT_READY)
632  return false;
633  // Update Future value chunk
634  *output_current_++ = chunk;
635  // Is that the last chunk?
636  --output_size_;
637  callback_output();
638  if (output_size_ == 0)

641  callback_status();
642  }

673  bool set_future_value_(const uint8_t* chunk, uint8_t size)
674  {
675  // Check size does not go beyond expected size
676  if (size > output_size_)
677  {
678  // Store error
679  set_future_error_(errors::EMSGSIZE);
680  return false;
681  }
682  while (size--)
683  {
684  if (!set_future_value_(*chunk++)) return false;
