666  template<board::DigitalPin CSN, board::DigitalPin CE> void NRF24L01<CSN, CE>::begin()
667  {
668  using namespace nrf24l01p_internals;
669  // Setup hardware features, channel, bitrate, retransmission, dynamic payload
670  write_register(Register::FEATURE, (bits::BV8(EN_DPL, EN_ACK_PAY, EN_DYN_ACK)));
671  write_register(Register::RF_CH, channel_);
672  write_register(Register::RF_SETUP, RF_DR_2MBPS | RF_PWR_0DBM);
673  write_register(Register::SETUP_RETR, uint8_t(DEFAULT_ARD << ARD) | uint8_t(DEFAULT_ARC << ARC));
674  write_register(Register::DYNPD, DPL_PA);
675 
676  // Setup hardware receive pipes address; network (16-bit), device (8-bit)
677  // P0: auto-acknowledge (see set_transmit_mode)
678  // P1: node address<network:device> with auto-acknowledge
679  // P2: broadcast<network:0>
680  addr_t rx_addr = addr_;
681  write_register(Register::SETUP_AW, AW_3BYTES);
682  write_register(Register::RX_ADDR_P1, (const uint8_t*) &rx_addr, sizeof(rx_addr));
683  write_register(Register::RX_ADDR_P2, BROADCAST);
684  write_register(Register::EN_RXADDR, bits::BV8(ERX_P2, ERX_P1));
685  write_register(Register::EN_AA, bits::BV8(ENAA_P1, ENAA_P0));
686 
687  // Ready to go
688  powerup();
689  }

691  template<board::DigitalPin CSN, board::DigitalPin CE> void NRF24L01<CSN, CE>::powerup()
692  {
693  using namespace nrf24l01p_internals;
694  if (state_ != State::POWER_DOWN_STATE) return;
695  ce_.clear();

725  template<board::DigitalPin CSN, board::DigitalPin CE>
726  int NRF24L01<CSN, CE>::send_(uint8_t dest, uint8_t port, const uint8_t* buf, size_t len)
727  {
728  using namespace nrf24l01p_internals;
729  // Note buf == 0 and len == 0 is perfectly acceptable
730  if ((buf == nullptr) && (len > 0)) return errors::EINVAL;
731  if (len > PAYLOAD_MAX) return errors::EMSGSIZE;
732 
733  // Setting transmit destination first (needs to ensure standby mode)
734  transmit_mode(dest);
735 
736  // Write source address and payload to the transmit fifo
737  Command command = ((dest != BROADCAST) ? Command::W_TX_PAYLOAD : Command::W_TX_PAYLOAD_NO_ACK);
738  this->start_transfer();
739  status_ = status_t{this->transfer(uint8_t(command))};
740  this->transfer(addr_.device);
741  this->transfer(port);
742  this->transfer(buf, len);
743  this->end_transfer();
744 
745  trans_ += 1;
746 
747  // Check for auto-acknowledge pipe(0), and address setup and enable
748  if (dest != BROADCAST)
749  {
750  addr_t tx_addr(addr_.network, dest);
751  write_register(Register::RX_ADDR_P0, (const uint8_t*) &tx_addr, sizeof(tx_addr));
752  write_register(Register::EN_RXADDR, bits::BV8(ERX_P2, ERX_P1, ERX_P0));
753  }
754 
755  // Wait for transmission
756  status_t status = status_t{0};
757  while (true)
758  {
759  status = read_status();
760  if (status.tx_ds || status.max_rt) break;
761  time::yield();
762  }
763 
764  bool data_sent = status.tx_ds;
765 
766  // Check for auto-acknowledge pipe(0) disable
767  if (dest != BROADCAST) write_register(Register::EN_RXADDR, bits::BV8(ERX_P2, ERX_P1));
768 
769  // Reset status bits
770  write_register(Register::STATUS, bits::BV8(TX_DS, MAX_RT));
771 
772  // Read retransmission counter and update
773  observe_tx_t observe = read_observe_tx();
774  retrans_ += observe.arc_cnt;
775 
776  // Check that the message was delivered
777  if (data_sent) return len;
778 
779  // Failed to deliver
780  write_command(Command::FLUSH_TX);
781  drops_ += 1;
782 
783  return errors::EIO;
784  }

786  template<board::DigitalPin CSN, board::DigitalPin CE>
787  int NRF24L01<CSN, CE>::recv_(uint8_t& src, uint8_t& port, uint8_t* buf, size_t size, uint32_t ms)
788  {
789  // Run in receive mode
790  receive_mode();
791 
792  // Check if there is data available on any pipe
793  uint32_t start = time::millis();
794  while (!available())
795  {
796  if ((ms != 0) && (time::since(start) > ms)) return errors::ETIME;
797  time::yield();
798  }
799 
800  // Try and read payload from FIFO
801  return read_fifo_payload(src, port, buf, size);
802  }

821  template<board::DigitalPin CSN, board::DigitalPin CE> void NRF24L01<CSN, CE>::transmit_mode(uint8_t dest)
822  {
823  using namespace nrf24l01p_internals;

825  addr_t tx_addr(addr_.network, dest);
826  write_register(Register::TX_ADDR, (const uint8_t*) &tx_addr, sizeof(tx_addr));

828  // Trigger the transmitter mode

854  template<board::DigitalPin CSN, board::DigitalPin CE> bool NRF24L01<CSN, CE>::available()
855  {
856  // Check the receiver fifo
857  if (read_fifo_status().rx_empty) return false;
858 
859  // Sanity check the size of the payload. Might require a flush
860  if (read_command(Command::R_RX_PL_WID) <= DEVICE_PAYLOAD_MAX) return true;
861  write_command(Command::FLUSH_RX);
862  return false;
863  }

865  template<board::DigitalPin CSN, board::DigitalPin CE>
866  int NRF24L01<CSN, CE>::read_fifo_payload(uint8_t& src, uint8_t& port, uint8_t* buf, size_t size)
867  {
868  // Check for payload error from device (Tab. 20, pp. 51, R_RX_PL_WID)
869  uint8_t count = read_command(Command::R_RX_PL_WID) - 2;
870  if ((count > PAYLOAD_MAX) || (count > size))
871  {
872  write_command(Command::FLUSH_RX);
873  return errors::EMSGSIZE;
874  }
875 
876  // Data is available, check if this a broadcast or not
877  dest_ = (read_status().rx_p_no == 1 ? addr_.device : BROADCAST);
878 
879  // Read the source address, port and payload
880  this->start_transfer();
881  status_ = status_t{this->transfer(uint8_t(Command::R_RX_PAYLOAD))};
882  src = this->transfer(0);
883  port = this->transfer(0);
884  this->transfer(buf, count, uint8_t(Command::NOP));
885  this->end_transfer();
886  return count;
887  }
