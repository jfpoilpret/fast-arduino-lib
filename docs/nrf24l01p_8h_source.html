590  template<board::DigitalPin CSN, board::DigitalPin CE> void NRF24L01<CSN, CE>::begin()
591  {
592  // Setup hardware features, channel, bitrate, retransmission, dynamic payload
593  write_register(Register::FEATURE, (_BV(EN_DPL) | _BV(EN_ACK_PAY) | _BV(EN_DYN_ACK)));
594  write_register(Register::RF_CH, channel_);
595  write_register(Register::RF_SETUP, RF_DR_2MBPS | RF_PWR_0DBM);
596  write_register(Register::SETUP_RETR, (DEFAULT_ARD << ARD) | (DEFAULT_ARC << ARC));
597  write_register(Register::DYNPD, DPL_PA);
598 
599  // Setup hardware receive pipes address; network (16-bit), device (8-bit)
600  // P0: auto-acknowledge (see set_transmit_mode)
601  // P1: node address<network:device> with auto-acknowledge
602  // P2: broadcast<network:0>
603  addr_t rx_addr = addr_;
604  write_register(Register::SETUP_AW, AW_3BYTES);
605  write_register(Register::RX_ADDR_P1, &rx_addr, sizeof(rx_addr));
606  write_register(Register::RX_ADDR_P2, BROADCAST);
607  write_register(Register::EN_RXADDR, _BV(ERX_P2) | _BV(ERX_P1));
608  write_register(Register::EN_AA, _BV(ENAA_P1) | _BV(ENAA_P0));
609 
610  // Ready to go
611  powerup();
612  }

614  template<board::DigitalPin CSN, board::DigitalPin CE> void NRF24L01<CSN, CE>::powerup()
615  {
616  if (state_ != State::POWER_DOWN_STATE) return;
617  ce_.clear();
618 
619  // Setup configuration for powerup and clear interrupts
620  write_register(Register::CONFIG, _BV(EN_CRC) | _BV(CRCO) | _BV(PWR_UP));
621  time::delay_ms(Tpd2stby_ms);
622  state_ = State::STANDBY_STATE;
623 
624  // Flush status
625  write_register(Register::STATUS, _BV(RX_DR) | _BV(TX_DS) | _BV(MAX_RT));
626  write_command(Command::FLUSH_TX);
627  write_command(Command::FLUSH_RX);
628  }

645  template<board::DigitalPin CSN, board::DigitalPin CE>
646  int NRF24L01<CSN, CE>::send(uint8_t dest, uint8_t port, const void* buf, size_t len)
647  {
648  if (buf == 0 && len > 0) return errors::EINVAL;
649  if (len > PAYLOAD_MAX) return errors::EMSGSIZE;
650 
651  // Setting transmit destination first (needs to ensure standby mode)
652  transmit_mode(dest);
653 
654  // Write source address and payload to the transmit fifo
655  Command command = ((dest != BROADCAST) ? Command::W_TX_PAYLOAD : Command::W_TX_PAYLOAD_NO_ACK);
656  this->start_transfer();
657  status_ = this->transfer(uint8_t(command));
658  this->transfer(addr_.device);
659  this->transfer(port);
660  this->transfer((uint8_t*) buf, len);
661  this->end_transfer();
662 
663  trans_ += 1;
664 
665  // Check for auto-acknowledge pipe(0), and address setup and enable
666  if (dest != BROADCAST)
667  {
668  addr_t tx_addr(addr_.network, dest);
669  write_register(Register::RX_ADDR_P0, &tx_addr, sizeof(tx_addr));
670  write_register(Register::EN_RXADDR, _BV(ERX_P2) | _BV(ERX_P1) | _BV(ERX_P0));
671  }
672 
673  // Wait for transmission
674  status_t status = 0;
675  while (true)
676  {
677  status = read_status();
678  if (status.tx_ds || status.max_rt) break;
679  time::yield();
680  }
681 
682  bool data_sent = status.tx_ds;
683 
684  // Check for auto-acknowledge pipe(0) disable
685  if (dest != BROADCAST) write_register(Register::EN_RXADDR, _BV(ERX_P2) | _BV(ERX_P1));
686 
687  // Reset status bits
688  write_register(Register::STATUS, _BV(TX_DS) | _BV(MAX_RT));
689 
690  // Read retransmission counter and update
691  observe_tx_t observe = read_observe_tx();
692  retrans_ += observe.arc_cnt;
693 
694  // Check that the message was delivered
695  if (data_sent) return len;
696 
697  // Failed to deliver
698  write_command(Command::FLUSH_TX);
699  drops_ += 1;
700 
701  return errors::EIO;
702  }

704  template<board::DigitalPin CSN, board::DigitalPin CE>
705  int NRF24L01<CSN, CE>::recv(uint8_t& src, uint8_t& port, void* buf, size_t size, uint32_t ms)
706  {
707  // Run in receive mode
708  receive_mode();
709 
710  // Check if there is data available on any pipe
711  uint32_t start = time::millis();
712  while (!available())
713  {
714  if ((ms != 0) && (time::since(start) > ms)) return errors::ETIME;
715  time::yield();
716  }
717 
718  // Try and read payload from FIFO
719  return read_fifo_payload(src, port, buf, size);
720  }

734  template<board::DigitalPin CSN, board::DigitalPin CE> void NRF24L01<CSN, CE>::transmit_mode(uint8_t dest)
735  {
736  // Setup primary transmit address
737  addr_t tx_addr(addr_.network, dest);
738  write_register(Register::TX_ADDR, &tx_addr, sizeof(tx_addr));
739 
740  // Trigger the transmitter mode
741  if (state_ != State::TX_STATE)
742  {
743  ce_.clear();
744  write_register(Register::CONFIG, _BV(EN_CRC) | _BV(CRCO) | _BV(PWR_UP));
745  ce_.set();
746  }
747 
748  // Wait for the transmitter to become active
749  if (state_ == State::STANDBY_STATE) time::delay_us(Tstby2a_us);
750  state_ = State::TX_STATE;
751  }

753  template<board::DigitalPin CSN, board::DigitalPin CE> void NRF24L01<CSN, CE>::receive_mode()
754  {
755  // Check already in receive mode
756  if (state_ == State::RX_STATE) return;
757 
758  // Configure primary receiver mode
759  write_register(Register::CONFIG, _BV(EN_CRC) | _BV(CRCO) | _BV(PWR_UP) | _BV(PRIM_RX));
760  ce_.set();
761  if (state_ == State::STANDBY_STATE) time::delay_us(Tstby2a_us);
762  state_ = State::RX_STATE;
763  }

765  template<board::DigitalPin CSN, board::DigitalPin CE> bool NRF24L01<CSN, CE>::available()
766  {
767  // Check the receiver fifo
768  if (read_fifo_status().rx_empty) return false;
769 
770  // Sanity check the size of the payload. Might require a flush
771  if (read_command(Command::R_RX_PL_WID) <= DEVICE_PAYLOAD_MAX) return true;
772  write_command(Command::FLUSH_RX);
773  return false;
774  }

776  template<board::DigitalPin CSN, board::DigitalPin CE>
777  int NRF24L01<CSN, CE>::read_fifo_payload(uint8_t& src, uint8_t& port, void* buf, size_t size)
778  {
779  // Check for payload error from device (Tab. 20, pp. 51, R_RX_PL_WID)
780  uint8_t count = read_command(Command::R_RX_PL_WID) - 2;
781  if ((count > PAYLOAD_MAX) || (count > size))
782  {
783  write_command(Command::FLUSH_RX);
784  return errors::EMSGSIZE;
785  }
786 
787  // Data is available, check if this a broadcast or not
788  dest_ = (read_status().rx_p_no == 1 ? addr_.device : BROADCAST);
789 
790  // Read the source address, port and payload
791  this->start_transfer();
792  status_ = this->transfer(uint8_t(Command::R_RX_PAYLOAD));
793  src = this->transfer(0);
794  port = this->transfer(0);
795  this->transfer((uint8_t*) buf, count, uint8_t(Command::NOP));
796  this->end_transfer();
797  return count;
798  }
799  }
