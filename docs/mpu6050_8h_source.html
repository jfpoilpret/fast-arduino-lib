864  class FifoPopFuture : public FUTURE<T, uint8_t>

866  using PARENT = FUTURE<T, uint8_t>;

873  bool get(T& result)
874  {
875  if (!PARENT::get(result)) return false;
876  // We need to swap bytes of all words (big-endian to little-endian)
877  uint16_t* temp = reinterpret_cast<uint16_t*>(&result);
878  uint8_t size = sizeof(T) / 2;
879  while (size--) utils::swap_bytes(*temp++);
880  return true;
881  }
