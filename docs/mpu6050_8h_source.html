579  bool fifo_pop(uint8_t* buffer, uint8_t size, bool wait, bool yield)
580  {
581  using namespace i2c::Status;
582  while (fifo_count() < size)
583  if (!wait)
584  return false;
585  else if (yield)
586  time::yield();
587 
588  if (this->write(DEVICE_ADDRESS, FIFO_R_W, BusCond::START_NO_STOP) == OK
589  && this->read(DEVICE_ADDRESS, buffer, size, BusCond::REPEAT_START_STOP) == OK)
590  {
591  // Swap all 2-bytes words
592  uint16_t* temp = (uint16_t*) buffer;
593  size /= 2;
594  while (size--) utils::swap_bytes(*temp++);
595  return true;
596  }
597  return false;
598  }
599  };
