157 static constexpr const uint8_t IOCON_INTPOL = bits::BV8(1);

159 static constexpr uint8_t compute_address(uint8_t address)
160 {
161 return uint8_t((uint8_t(BASE_ADDRESS) | uint8_t(address & 0x07U)) << 1);
162 }
