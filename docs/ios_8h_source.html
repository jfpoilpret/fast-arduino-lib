547  static size_t double_digits(double v)
548  {
549  int digits = int(log10(fabs(v)));
550  if (digits < 0)
551  return 1;
552  else
553  return size_t(digits + 1);
554  }
555  bool is_too_large(double v) const
556  {
557  // Number of chars = sign + digits before DP + DP + precision + \0
558  return (1 + double_digits(v) + 1 + precision() + 1 > DOUBLE_BUFFER_SIZE);
559  }
560  void convert(ostreambuf& out, double v) const
561  {
562  // Allocate sufficient size for fixed/scientific representation with precision max = 16
563  // Need 1 more for sign, 1 for DP, 1 for first digit, 4 for e+00
564  char buffer[DOUBLE_BUFFER_SIZE];
565  // If v is too large, force scientific anyway
566  if ((flags() & scientific) || is_too_large(v))
567  {
568  const uint8_t DTOSTRE_MAX_PRECISION = 7;
569  dtostre(v, buffer, precision(), 0);
570  if (precision() > DTOSTRE_MAX_PRECISION)
571  {
572  // If precision() > 7, then it is limited to 7 by dtostre(), add 0 manually then
573  // Size of the exponent part, always e+00 or e-00
574  const uint8_t EXPONENT_SIZE = 1 + 1 + 2;
575  // Find exponent marker 'e' and move exponent part right (including null)
576  char* exp = strchr(buffer, 'e');
577  uint8_t added_zeros = precision() - DTOSTRE_MAX_PRECISION;
578  memmove(exp + added_zeros, exp, EXPONENT_SIZE + 1);
579  // Add '0' in the added space before exponent
580  memset(exp, '0', added_zeros);
581  }
582  }
583  else if (flags() & fixed)
584  dtostrf(v, 0, precision(), buffer);
585  else
586  {
587  // In this mode any trailing 0 after DP is discarded
588  // DP is also discarded if there are only 0 after it
589  dtostrf(v, 0, precision(), buffer);
590  char* dp = strchr(buffer, '.');
591  if (dp)
592  {
593  // Number has a decimal point, remove all trailing zeros
594  char* reverse = buffer + strlen(buffer) - 1;
595  while (reverse != dp && *reverse == '0')
596  *reverse-- = 0;
597  // If there is no significant digit after DP, remove DP
598  if (reverse == dp)
599  *reverse = 0;
600  }
601  }
602  upper(buffer, true);
603  justify(out, buffer, add_sign(buffer, true), 0);
604  }

657  void justify(ostreambuf& out, const char* input, bool add_sign, const char* prefix) const
658  {
659  // Handle case where padding must be added
660  // Speed optimization: handle padding situation ONLY if width is != 0
661  if (width())
662  {
663  size_t len = strlen(input) + (prefix ? strlen(prefix) : 0) + (add_sign ? 1 : 0);
664  if (len < width())
665  {
666  uint8_t add = width() - len;
667  if (flags() & left)
668  {
669  output_number(out, input, add_sign, prefix);
670  output_filler(out, fill(), add);
671  out.on_put();
672  }
673  else
674  {
675  output_filler(out, fill(), add);
676  output_number(out, input, add_sign, prefix);
677  }
678  }
679  return;
680  }
681  // Handle case where no padding is needed
682  output_number(out, input, add_sign, prefix);
683  }
