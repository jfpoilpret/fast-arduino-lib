138  class WatchdogSignal
139  {
140  public:
147  void begin(TimeOut timeout = TimeOut::TO_16ms)
148  {
149  uint8_t config = _BV(WDIE) | (uint8_t(timeout) & 0x07) | (uint8_t(timeout) & 0x08 ? _BV(WDP3) : 0);
150  synchronized begin_with_config(config);
151  }
152 
156  void end()
157  {
158  synchronized
159  {
160  WDTCSR_ = _BV(WDCE) | _BV(WDE);
161  WDTCSR_ = 0;
162  }
163  }
164 
165  protected:
167  inline void begin_with_config(uint8_t config) INLINE
168  {
169  __asm__ __volatile__("wdr");
170  MCUSR_ |= 1 << WDRF;
171  WDTCSR_ = _BV(WDCE) | _BV(WDE);
172  WDTCSR_ = config;
173  }
175 
176  private:
177  using REG8 = board_traits::REG8;
178  static constexpr const REG8 MCUSR_{MCUSR};
179  static constexpr const REG8 WDTCSR_{WDTCSR};
180  };
