575  template<typename MANAGER> class I2CFuturesGroup :
576  public AbstractI2CFuturesGroup<MANAGER>
577  {
578  using PARENT = AbstractI2CFuturesGroup<MANAGER>;
579  using MANAGER_TRAIT = I2CManager_trait<MANAGER>;
580  using STATUS_LISTENER = typename PARENT::STATUS_LISTENER;
581  using ABSTRACT_FUTURE = typename PARENT::ABSTRACT_FUTURE;
582 
583  protected:
592  I2CFuturesGroup(
593  ABSTRACT_FUTURE** futures, uint8_t size, STATUS_LISTENER* status_listener = nullptr)
594  : PARENT{status_listener}, futures_{futures}, size_{size} {}
595 
610  bool start(typename PARENT::DEVICE& device)
611  {
612  PARENT::set_device(device);
613  if (MANAGER_TRAIT::IS_ASYNC)
614  {
615  return next_future();
616  }
617  else
618  {
619  // In sync mode, we replace recursive calls (generated by on_status_change()) by a loop
620  while (index_ != size_)
621  {
622  if (!next_future()) return false;
623  }
624  return true;
625  }
626  }
627 
629  void on_status_change(const ABSTRACT_FUTURE& future, future::FutureStatus status) final
630  {
631  PARENT::on_status_change(future, status);
632  // In sync mode, we must avoid recursive calls generated by on_status_change()!
633  if (MANAGER_TRAIT::IS_ASYNC)
634  {
635  // First check that current future was executed successfully
636  if (status == future::FutureStatus::READY)
637  next_future();
638  }
639  }
641 
642  private:
643  bool next_future()
644  {
645  if (index_ == size_)
646  // Future is finished
647  return false;
648 
649  ABSTRACT_FUTURE& future = *futures_[index_++];
650  // Check if future has read, write or both
651  const bool stop = (index_ == size_);
652  const bool read = future.get_future_value_size_();
653  const bool write = future.get_storage_value_size_();
654  int error = 0;
655  if (read && write)
656  {
657  error = PARENT::launch_commands(future, {PARENT::write(), PARENT::read(0, false, stop)});
658  }
659  else if (read)
660  {
661  error = PARENT::launch_commands(future, {PARENT::read(0, false, stop)});
662  }
663  else if (write)
664  {
665  error = PARENT::launch_commands(future, {PARENT::write(0, false, stop)});
666  }
667  else
668  {
669  //FIXME we consider that any other future is an I2CFuturesGroup, which might not always be correct!
670  I2CFuturesGroup& group = static_cast<I2CFuturesGroup&>(future);
671  if (!group.start(PARENT::device()))
672  error = errors::EILSEQ;
673  }
674  return PARENT::check_error(error);
675  }
676 
677  ABSTRACT_FUTURE** futures_;
678  uint8_t size_;
679  uint8_t index_ = 0;
680  };

696  template<typename MANAGER> class I2CSameFutureGroup : public AbstractI2CFuturesGroup<MANAGER>
697  {
698  using PARENT = AbstractI2CFuturesGroup<MANAGER>;
699  using MANAGER_TRAIT = I2CManager_trait<MANAGER>;
700  using STATUS_LISTENER = typename PARENT::STATUS_LISTENER;
701  using ABSTRACT_FUTURE = typename PARENT::ABSTRACT_FUTURE;
702  using F = WriteRegisterFuture<MANAGER, uint8_t>;
703  using CONTENT = WriteContent<uint8_t>;
704  static constexpr uint8_t FUTURE_SIZE = F::IN_SIZE;
705 
706  public:
715  I2CSameFutureGroup(uint16_t address, uint8_t size, STATUS_LISTENER* status_listener = nullptr)
716  : PARENT{status_listener}, address_{address}, size_{size}
717  {
718  PARENT::init({&future_}, size / FUTURE_SIZE);
719  }
720 
735  bool start(typename PARENT::DEVICE& device)
736  {
737  PARENT::set_device(device);
738  if (MANAGER_TRAIT::IS_ASYNC)
739  {
740  return next_future();
741  }
742  else
743  {
744  // In sync mode, we replace recursive calls (generated by on_status_change()) by a loop
745  while (size_)
746  {
747  if (!next_future()) return false;
748  }
749  return true;
750  }
751  }
752 
753  private:
754  bool next_future()
755  {
756  if (size_ == 0)
757  // Future is finished already
758  return false;
759 
760  uint8_t reg = next_byte();
761  uint8_t val = next_byte();
762  const bool stop = (size_ == 0);
763  future_.reset_(CONTENT{reg, val});
764  return PARENT::check_error(
765  PARENT::launch_commands(future_, {PARENT::write(0, false, stop)}));
766  }
767 
768  // Get the next byte, from the flash
769  uint8_t next_byte()
770  {
771  uint8_t data = 0;
772  --size_;
773  return flash::read_flash(address_++, data);
774  }
775 
776  void on_status_change(const ABSTRACT_FUTURE& future, future::FutureStatus status) final
777  {
778  PARENT::on_status_change(future, status);
779  // In sync mode, we must avoid recursive calls generated by on_status_change()!
780  if (MANAGER_TRAIT::IS_ASYNC)
781  {
782  // First check that current future was executed successfully
783  if (status == future::FutureStatus::READY)
784  next_future();
785  }
786  }
787 
788  // Address of flah mempry holding information about bytes to write
789  uint16_t address_;
790  uint8_t size_;
791  // The future reused for all writes
792  F future_{0, 0};
793
