218  template<typename CLOCK, typename T> bool Scheduler<CLOCK, T>::operator()(Job& job)
219  {
220  uint32_t now = clock_.millis();
221  if (job.next_time() <= now)
222  {
223  job.on_schedule(now);
224  if (!job.is_periodic()) return true;
225  job.reschedule(now + job.period());
226  }
227  return false;
