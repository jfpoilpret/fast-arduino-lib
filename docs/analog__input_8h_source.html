100  SAMPLE_TYPE sample()
101  {
102  // First ensure that any pending sampling is finished
103  GLOBAL_TRAIT::ADCSRA_.loop_until_bit_clear(ADSC);
104  // Setup multiplexer selection and start conversion
105  GLOBAL_TRAIT::ADMUX_ = AREF_TRAIT::MASK | TYPE_TRAIT::ADLAR1 | TRAIT::MUX_MASK1;
106  GLOBAL_TRAIT::ADCSRB_ = TRAIT::MUX_MASK2 | TYPE_TRAIT::ADLAR2;
107 
108  // The following delay is necessary for bandgap ADC, strangely 70us should be enough (datasheet)
109  // but this works only when no other ADC is used "at the same time"
110  // In this situation, a delay of minimum 400us seems necessary to ensure bandgap reference voltage is stabilized
111  if (TRAIT::IS_BANDGAP) time::delay_us(BG_STABILIZATION_DELAY_US);
112 
113  GLOBAL_TRAIT::ADCSRA_ = _BV(ADEN) | _BV(ADSC) | TRAIT::MUX_MASK2 | FREQ_TRAIT::PRESCALER_MASK;
114  // Wait until sampling is done
115  GLOBAL_TRAIT::ADCSRA_.loop_until_bit_clear(ADSC);
116  // Should we synchronize ADC reading?
117  return TYPE_TRAIT::ADC_;
118  }
