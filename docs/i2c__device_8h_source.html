254  int launch_commands(
255  PROXY<ABSTRACT_FUTURE> proxy, utils::range<I2CLightCommand> commands)
256  {
257  uint8_t num_commands = commands.size();
258  if (num_commands == 0) return errors::EINVAL;
259  {
260  // Truly asynchronous mode (ATmega only)
261  // The whole code block in launch_commands() must be synchronized
262  UNUSED auto outer_sync = DisableInterrupts<MANAGER_TRAIT::IS_ASYNC>{};
263  uint8_t max_read;
264  uint8_t max_write;
265  {
266  // Synchronous mode
267  // Only the next few method calls shall be synchronized
268  UNUSED auto inner_sync = DisableInterrupts<!MANAGER_TRAIT::IS_ASYNC>{};
269  // pre-conditions (must be synchronized)
270  if (!handler_.ensure_num_commands_(num_commands)) return errors::EAGAIN;
271  ABSTRACT_FUTURE& future = resolve(proxy);
272  max_read = future.get_future_value_size_();
273  max_write = future.get_storage_value_size_();
274  }
275 
276  // That check is normally usefull only in debug mode
277  if (MANAGER_TRAIT::IS_DEBUG && (!check_commands(max_write, max_read, commands)))
278  return errors::EINVAL;
279 
280  // Now push each command to the I2C Manager
281  int error = 0;
282  for (I2CLightCommand command : commands)
283  {
284  // update command.byte_count if 0
285  command.update_byte_count(max_read, max_write);
286  // force future finish for last command in transaction
287  --num_commands;
288  if (num_commands == 0)
289  command.type().add_flags(auto_stop_flags_);
290  // Note: on ATtiny, this method blocks until I2C command is finished!
291  if (!handler_.push_command_(command, device_, proxy))
292  {
293  error = errors::EPROTO;
294  break;
295  }
296  }
297  // Notify handler that transaction is complete
298  handler_.last_command_pushed_();
299  return error;
300  }
301  }

471  private:
472  static bool check_commands(
473  uint8_t max_write, uint8_t max_read, const utils::range<I2CLightCommand>& commands)
474  {
475  // Limit total number of bytes read or written in a transaction to 255
476  uint8_t total_read = 0;
477  uint8_t total_write = 0;
478  for (const I2CLightCommand& command : commands)
479  {
480  // Count number of bytes read and written
481  if (command.type().is_write())
482  total_write += (command.byte_count() ? command.byte_count() : max_write);
483  else
484  total_read += (command.byte_count() ? command.byte_count() : max_read);
485  }
486  // check sum of read commands byte_count matches future output size
487  // check sum of write commands byte_count matches future input size
488  return (total_write == max_write) && (total_read == max_read);
489  }
