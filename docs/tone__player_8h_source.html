369  static constexpr uint32_t period(Tone tone)
370  {
371  return ONE_SECOND / 2 / uint16_t(tone);

373  static constexpr PRESCALER prescaler(Tone tone)
374  {
375  return (tone > Tone::SILENCE ? CALC::CTC_prescaler(period(tone)) : PRESCALER::NO_PRESCALING);
376  }

378  {
379  return (tone > Tone::SILENCE ? CALC::CTC_counter(prescaler(tone), period(tone)) : 0);

386  if (tone == SpecialTone::REPEAT_END) return REPEAT_END;
387  if (tone == SpecialTone::TIE) return TIE;
388  return TONE;

572  void prepare_(const TONE_PLAY* melody, LOAD_TONE load_tone)
573  {
574  loader_ = load_tone;
575  current_play_ = melody;
576  repeat_play_ = nullptr;
577  repeat_times_ = 0;
578  tie_notes_ = 0;
579  }
580 
581  void reset_()
582  {
583  loader_ = nullptr;
584  current_play_ = nullptr;
585  repeat_play_ = nullptr;
586  repeat_times_ = 0;
587  tie_notes_ = 0;
588  }
589 
590  uint16_t start_next_()
591  {
592  if (loader_ == nullptr) return 0;
593 
594  TONE_PLAY holder;
595  const TONE_PLAY* current = loader_(current_play_, holder);
596  if (current->is_end())
597  {
598  reset_();
599  return 0;
600  }
601  uint16_t delay = 0;
602  no_delay_ = true;
603  if (current->is_repeat_start())
604  {
605  repeat_play_ = current_play_;
606  repeat_times_ = -1;
607  }
608  else if (current->is_repeat_end())
609  {
610  if (repeat_play_)
611  {
612  if (repeat_times_ == -1) repeat_times_ = current->repeat_count();
613  if (repeat_times_--)
614  current_play_ = repeat_play_;
615  else
616  repeat_play_ = nullptr;
617  }
618  }
619  else if (current->is_tie())
620  {
621  tie_notes_ = current->num_ties();
622  }
623  else
624  {
625  if (current->is_tone()) current->generate_tone(generator_);
626  if (tie_notes_)
627  --tie_notes_;
628  else
629  no_delay_ = false;
630  delay = duration(current->duration());
631  }
632  ++current_play_;
633  return delay;
634  }
635 
636  uint16_t stop_current_()
637  {
638  if (no_delay_)
639  return 0;
640  generator_.stop_tone();
641  return INTERTONE_DELAY_MS;
642  }
643 
644  uint16_t duration(Duration d) const
645  {
646  return uint8_t(d) * t32_duration_ms_;
647  }

661  return &holder;
662  }
663 
664  GENERATOR& generator_;
665  LOAD_TONE loader_ = nullptr;
666  uint16_t t32_duration_ms_ = 0U;
667  const TONE_PLAY* current_play_ = nullptr;
668  const TONE_PLAY* repeat_play_ = nullptr;
669  int8_t repeat_times_ = 0;
670  uint8_t tie_notes_ = 0;
671  bool no_delay_ = false;
672  };
