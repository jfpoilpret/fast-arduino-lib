172  template<typename OSTREAM>
173  void trace(OSTREAM& out)
174  {
175  for (uint8_t i = 0; i < index_; ++i)
176  {
177  const i2c::DebugStatus status = status_[i];
178  switch (status)
179  {
180  case i2c::DebugStatus::START:
181  case i2c::DebugStatus::REPEAT_START:
182  case i2c::DebugStatus::STOP:
183  case i2c::DebugStatus::RECV_ERROR:
184  case i2c::DebugStatus::RECV:
185  case i2c::DebugStatus::RECV_LAST:
186  case i2c::DebugStatus::SEND_ERROR:
187  case i2c::DebugStatus::SEND_OK:
188  out << status;
189  break;
190 
191  default:
192  out << status;
193  out.setf(streams::ios::hex, streams::ios::basefield);
194  out << data_[i] << ' ';
195  }
196  }
197  if (index_ >= SIZE)
198  out << F("# OVF #");
199  out << '\n';
200  index_ = 0;
201  }
202 
204  void operator()(i2c::DebugStatus status, uint8_t data)
205  {
206  if (index_ >= SIZE) return;
207  switch (status)
208  {
209  case i2c::DebugStatus::START:
210  case i2c::DebugStatus::REPEAT_START:
211  case i2c::DebugStatus::STOP:
212  case i2c::DebugStatus::SLAW:
213  case i2c::DebugStatus::SLAR:
214  case i2c::DebugStatus::SEND:
215  case i2c::DebugStatus::RECV:
216  case i2c::DebugStatus::RECV_LAST:
217  if (uint8_t(debug_) & uint8_t(DEBUG::DEBUG_STEPS))
218  {
219  status_[index_] = status;
220  data_[index_++] = data;
221  }
222  break;
223 
224  case i2c::DebugStatus::SEND_OK:
225  if (uint8_t(debug_) & uint8_t(DEBUG::DEBUG_SEND_OK))
226  {
227  status_[index_] = status;
228  data_[index_++] = data;
229  }
230  break;
231 
232  case i2c::DebugStatus::SEND_ERROR:
233  if (uint8_t(debug_) & uint8_t(DEBUG::DEBUG_SEND_ERR))
234  {
235  status_[index_] = status;
236  data_[index_++] = data;
237  }
238  break;
239 
240  case i2c::DebugStatus::RECV_OK:
241  if (uint8_t(debug_) & uint8_t(DEBUG::DEBUG_RECV_OK))
242  {
243  status_[index_] = status;
244  data_[index_++] = data;
245  }
246  break;
247 
248  case i2c::DebugStatus::RECV_ERROR:
249  if (uint8_t(debug_) & uint8_t(DEBUG::DEBUG_RECV_ERR))
250  {
251  status_[index_] = status;
252  data_[index_++] = data;
253  }
254  break;
255  }
256  }
258 
259  private:
260  i2c::DebugStatus status_[SIZE];
261  uint8_t data_[SIZE];
262  uint8_t index_ = 0;
263  DEBUG debug_;
264  };
