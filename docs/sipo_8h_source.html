129 void bit_bang_data(uint8_t data)
130 {
131 // Start with MSB
132 uint8_t mask = 0x80;
133 do
134 {
135 if (data & mask)
136 data_.set();
137 else
138 data_.clear();
139 clock_.set();
140 mask >>= 1;
141 clock_.clear();
142 } while (mask);
