131  template<I2CErrorPolicy POLICY = I2CErrorPolicy::DO_NOTHING> struct I2CErrorPolicySupport
132  {
133  I2CErrorPolicySupport() = default;
134  template<typename T>
135  void handle_error(UNUSED const I2CCommand<T>& current, UNUSED containers::Queue<I2CCommand<T>>& commands)
136  {
137  // Intentionally empty: do nothing in this policy
138  }
139  };
140  template<> struct I2CErrorPolicySupport<I2CErrorPolicy::CLEAR_ALL_COMMANDS>
141  {
142  I2CErrorPolicySupport() = default;
143  template<typename T>
144  void handle_error(UNUSED const I2CCommand<T>& current, containers::Queue<I2CCommand<T>>& commands)
145  {
146  commands.clear_();

149  template<> struct I2CErrorPolicySupport<I2CErrorPolicy::CLEAR_TRANSACTION_COMMANDS>
150  {
151  I2CErrorPolicySupport() = default;
152  template<typename T>
153  void handle_error(const I2CCommand<T>& current, containers::Queue<I2CCommand<T>>& commands)
154  {
155  // Clear command belonging to the same transaction (i.e. same future)
156  const auto future = current.future();
157  I2CCommand<T> command;
158  while (commands.peek_(command))
159  {
160  if (command.future() != future)
161  break;
162  commands.pull_(command);
163  }
164  }
165  };

172  template<I2CMode MODE_, bool HAS_STATUS_ = false, typename STATUS_HOOK_ = I2C_STATUS_HOOK>
173  class ATmegaI2CSyncHandler
174  {
175  private:
176  using MODE_TRAIT = I2CMode_trait<MODE_>;
177  using I2C_TRAIT = board_traits::TWI_trait;
178  using REG8 = board_traits::REG8;
179  using STATUS = I2CStatusSupport<HAS_STATUS_, STATUS_HOOK_>;
180 
181  public:
182  explicit ATmegaI2CSyncHandler(STATUS_HOOK_ status_hook = nullptr) : status_hook_{status_hook} {}
183 
184  void begin_()
185  {
186  // 1. set SDA/SCL pullups
187  I2C_TRAIT::PORT |= I2C_TRAIT::SCL_SDA_MASK;
188  // 2. set I2C frequency
189  TWBR_ = MODE_TRAIT::FREQUENCY;
190  TWSR_ = 0;
191  // 3. Enable TWI
192  TWCR_ = bits::BV8(TWEN);
193  }
194 
195  void end_()
196  {
197  // 1. Disable TWI
198  TWCR_ = 0;
199  // 2. remove SDA/SCL pullups
200  I2C_TRAIT::PORT &= bits::COMPL(I2C_TRAIT::SCL_SDA_MASK);
201  }
202 
203  // Low-level methods to handle the bus in an asynchronous way
204  bool exec_start_()
205  {
206  TWCR_ = bits::BV8(TWEN, TWINT, TWSTA);
207  return wait_twint(Status::START_TRANSMITTED);
208  }
209 
210  bool exec_repeat_start_()
211  {
212  TWCR_ = bits::BV8(TWEN, TWINT, TWSTA);
213  return wait_twint(Status::REPEAT_START_TRANSMITTED);
214  }
215 
216  bool exec_send_slar_(uint8_t target)
217  {
218  send_byte(target | 0x01U);
219  return wait_twint(Status::SLA_R_TRANSMITTED_ACK);
220  }
221 
222  bool exec_send_slaw_(uint8_t target)
223  {
224  send_byte(target);
225  return wait_twint(Status::SLA_W_TRANSMITTED_ACK);
226  }
227 
228  bool exec_send_data_(uint8_t data)
229  {
230  send_byte(data);
231  return wait_twint(Status::DATA_TRANSMITTED_ACK);
232  }
233 
234  bool exec_receive_data_(bool last_byte, uint8_t& data)
235  {
236  const uint8_t twcr = (last_byte ? bits::BV8(TWEN, TWINT) : bits::BV8(TWEN, TWINT, TWEA));
237  const Status expected = (last_byte ? Status::DATA_RECEIVED_NACK : Status::DATA_RECEIVED_ACK);
238  TWCR_ = twcr;
239  if (wait_twint(expected))
240  {
241  data = TWDR_;
242  return true;
243  }
244  return false;
245  }
246 
247  void exec_stop_()
248  {
249  TWCR_ = bits::BV8(TWEN, TWINT, TWSTO);
250  }
251 
252  private:
253  static constexpr const REG8 TWBR_{TWBR};
254  static constexpr const REG8 TWSR_{TWSR};
255  static constexpr const REG8 TWCR_{TWCR};
256  static constexpr const REG8 TWDR_{TWDR};
257 
258  void send_byte(uint8_t data)
259  {
260  TWDR_ = data;
261  TWCR_ = bits::BV8(TWEN, TWINT);
262  }
263 
264  bool wait_twint(Status expected_status)
265  {
266  TWCR_.loop_until_bit_set(TWINT);
267  const Status status = Status(TWSR_ & bits::BV8(TWS3, TWS4, TWS5, TWS6, TWS7));
268  status_hook_.call_hook(expected_status, status);
269  return (status == expected_status);
270  }
271 
272  STATUS status_hook_;
273  };

478  // States of execution of an I2C command through ISR calls
479  enum class State : uint8_t
480  {
481  NONE = 0,
482  START,
483  SLAW,
484  SLAR,
485  SEND,
486  RECV,
487  RECV_LAST,
488  STOP
489  };
490 
491  ABSTRACT_FUTURE& current_future() const
492  {
493  return command_.future();
494  }
495 
496  void send_byte(uint8_t data)
497  {
498  TWDR_ = data;
499  TWCR_ = bits::BV8(TWEN, TWIE, TWINT);
500  }
501 
502  // Dequeue the next command in the queue and process it immediately
503  void dequeue_command_(bool first)
504  {
505  if (!commands_.pull_(command_))
506  {
507  command_ = I2CCOMMAND{};
508  current_ = State::NONE;
509  // No more I2C command to execute
510  TWCR_ = bits::BV8(TWINT);
511  return;
512  }
513 
514  // Start new commmand
515  current_ = State::START;
516  if (first)
517  exec_start_();
518  else
519  exec_repeat_start_();
520  }
521 
522  // Method to compute next state
523  State next_state_()
524  {
525  switch (current_)
526  {
527  case State::START:
528  return (command_.type().is_write() ? State::SLAW : State::SLAR);
529 
530  case State::SLAR:
531  case State::RECV:
532  if (command_.byte_count() > 1)
533  return State::RECV;
534  else
535  return State::RECV_LAST;
536 
537  case State::RECV_LAST:
538  return State::STOP;
539 
540  case State::SLAW:
541  return State::SEND;
542 
543  case State::SEND:
544  if (command_.byte_count() >= 1)
545  return State::SEND;
546  else
547  return State::STOP;
548 
549  case State::STOP:
550  case State::NONE:
551  return State::NONE;
552  }
553  }
554 
555  // Low-level methods to handle the bus in an asynchronous way
556  void exec_start_()
557  {
558  debug_hook_.call_hook(DebugStatus::START);
559  expected_status_ = Status::START_TRANSMITTED;
560  TWCR_ = bits::BV8(TWEN, TWIE, TWINT, TWSTA);
561  }
562  void exec_repeat_start_()
563  {
564  debug_hook_.call_hook(DebugStatus::REPEAT_START);
565  expected_status_ = Status::REPEAT_START_TRANSMITTED;
566  TWCR_ = bits::BV8(TWEN, TWIE, TWINT, TWSTA);
567  }
568  void exec_send_slar_()
569  {
570  debug_hook_.call_hook(DebugStatus::SLAR, command_.target());
571  // Read device address from queue
572  expected_status_ = Status::SLA_R_TRANSMITTED_ACK;
573  send_byte(command_.target() | 0x01U);
574  }
575  void exec_send_slaw_()
576  {
577  debug_hook_.call_hook(DebugStatus::SLAW, command_.target());
578  // Read device address from queue
579  expected_status_ = Status::SLA_W_TRANSMITTED_ACK;
580  send_byte(command_.target());
581  }
582  void exec_send_data_()
583  {
584  // Determine next data byte
585  uint8_t data = 0;
586  ABSTRACT_FUTURE& future = current_future();
587  bool ok = future.get_storage_value_(data);
588  debug_hook_.call_hook(DebugStatus::SEND, data);
589  // This should only happen if there are 2 concurrent consumers for that Future
590  if (ok)
591  command_.decrement_byte_count();
592  else
593  future.set_future_error_(errors::EILSEQ);
594  debug_hook_.call_hook(ok ? DebugStatus::SEND_OK : DebugStatus::SEND_ERROR);
595  expected_status_ = Status::DATA_TRANSMITTED_ACK;
596  send_byte(data);
597  }
598  void exec_receive_data_()
599  {
600  // Is this the last byte to receive?
601  if (command_.byte_count() == 1)
602  {
603  debug_hook_.call_hook(DebugStatus::RECV_LAST);
604  // Send NACK for the last data byte we want
605  expected_status_ = Status::DATA_RECEIVED_NACK;
606  TWCR_ = bits::BV8(TWEN, TWIE, TWINT);
607  }
608  else
609  {
610  debug_hook_.call_hook(DebugStatus::RECV);
611  // Send ACK for data byte if not the last one we want
612  expected_status_ = Status::DATA_RECEIVED_ACK;
613  TWCR_ = bits::BV8(TWEN, TWIE, TWINT, TWEA);
614  }
615  }
616  void exec_stop_(bool error = false)
617  {
618  debug_hook_.call_hook(DebugStatus::STOP);
619  TWCR_ = bits::BV8(TWEN, TWINT, TWSTO);
620  if (!error)
621  expected_status_ = Status::OK;
622  command_ = I2CCOMMAND{};
623  current_ = State::NONE;
624  // If so then delay 4.0us + 4.7us (100KHz) or 0.6us + 1.3us (400KHz)
625  // (ATMEGA328P datasheet 29.7 Tsu;sto + Tbuf)
626  _delay_loop_1(MODE_TRAIT::DELAY_AFTER_STOP);
627  }
628 
629  bool is_end_transaction() const
630  {
631  return command_.type().is_end();
632  }
633 
634  bool handle_no_error(ABSTRACT_FUTURE& future, Status status)
635  {
636  if (check_no_error(future, status)) return true;
637  policy_.handle_error(command_, commands_);
638  // In case of an error, immediately send a STOP condition
639  exec_stop_(true);
640  dequeue_command_(true);
641  return false;
642  }

644  I2CCallback i2c_change()
645  {
646  // Check status Vs. expected status
647  const Status status = Status(TWSR_ & bits::BV8(TWS3, TWS4, TWS5, TWS6, TWS7));
648  ABSTRACT_FUTURE& future = current_future();
649  if (!handle_no_error(future, status))
650  return I2CCallback::ERROR;
651 
652  // Handle TWI interrupt when data received
653  if ((current_ == State::RECV) || (current_ == State::RECV_LAST))
654  {
655  const uint8_t data = TWDR_;
656  bool ok = future.set_future_value_(data);
657  // This should only happen in case there are 2 concurrent providers for this future
658  if (ok)
659  command_.decrement_byte_count();
660  else
661  future.set_future_error_(errors::EILSEQ);
662  debug_hook_.call_hook(ok ? DebugStatus::RECV_OK : DebugStatus::RECV_ERROR, data);
663  }
664 
665  // Handle next step in current command
666  I2CCallback result = I2CCallback::NONE;
667  current_ = next_state_();
668  switch (current_)
669  {
670  case State::NONE:
671  case State::START:
672  // This cannot happen
673  break;
674 
675  case State::SLAR:
676  exec_send_slar_();
677  break;
678 
679  case State::RECV:
680  case State::RECV_LAST:
681  exec_receive_data_();
682  break;
683 
684  case State::SLAW:
685  exec_send_slaw_();
686  break;
687 
688  case State::SEND:
689  exec_send_data_();
690  break;
691 
692  case State::STOP:
693  // Check if we need to finish the current future
694  if (command_.type().is_finish())
695  future.set_future_finish_();
696  result = (is_end_transaction() ? I2CCallback::END_TRANSACTION : I2CCallback::END_COMMAND);
697  // Check if we need to STOP (no more pending commands in queue)
698  if (commands_.empty_())
699  exec_stop_();
700  // Check if we need to STOP or REPEAT START (current command requires STOP)
701  else if (command_.type().is_stop())
702  {
703  exec_stop_();
704  // Handle next command
705  dequeue_command_(true);
706  }
707  else
708  // Handle next command
709  dequeue_command_(false);
710  }
711  return result;
712  }
713 
714  bool check_no_error(ABSTRACT_FUTURE& future, Status status)
715  {
716  status_hook_.call_hook(expected_status_, status);
717  if (status == expected_status_) return true;
718  // Handle special case of last transmitted byte possibly not acknowledged by device
719  if ( (expected_status_ == Status::DATA_TRANSMITTED_ACK)
720  && (status == Status::DATA_TRANSMITTED_NACK)
721  && (command_.byte_count() == 0))
722  return true;
723 
724  // The future must be marked as error
725  if (future.status() != future::FutureStatus::ERROR)
726  future.set_future_error_(errors::EPROTO);
727  return false;
728  }
729 
730  // Status of current command processing
731  I2CCOMMAND command_;
732 
733  // Latest I2C status
734  Status expected_status_ = Status::OK;
735 
736  // Status of current command processing
737  State current_ = State::NONE;
738 
739  // Queue of commands to execute
740  containers::Queue<I2CCOMMAND> commands_;
741 
742  POLICY policy_{};
743  STATUS status_hook_;
744  DEBUG debug_hook_;
